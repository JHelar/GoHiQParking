CREATE TABLE Users (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
Email TEXT NON NULL,
Username TEXT NON NULL,
Salt TEXT NON NULL,
Password TEXT NON NULL,
Created TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
IsDisabled BOOLEAN DEFAULT FALSE
);
CREATE TABLE UserSessions (
SessionKey TEXT PRIMARY KEY,
UserID INTEGER NOT NULL,
LoginTime TIMESTAMP NOT NULL,
LastSeenTime TIMESTAMP NOT NULL,
FOREIGN KEY(UserID) REFERENCES Users(ID)
);
CREATE TABLE ParkingLots (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
Name TEXT,
Location TEXT
);
CREATE TABLE Spots (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
Name TEXT,
IsParked BOOLEAN,
ParkedBy INTEGER, 
ParkedTime TIMESTAMP, 
ParkingLot INTEGER,
FOREIGN KEY(ParkedBy) REFERENCES Users(ID)
FOREIGN KEY(ParkingLot) REFERENCES ParkingLots(ID)
);
CREATE TABLE SpotInfos (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
SpotID INTEGER,
Description TEXT,
Longitude REAL,
Latitude REAL,
ImagePath TEXT,
FOREIGN KEY(SpotID) REFERENCES Spots(ID)
);
